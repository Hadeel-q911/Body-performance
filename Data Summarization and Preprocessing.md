##Boxplot of age
```{r}
boxplot.stats(dataset$age)$out

out <- boxplot.stats(dataset$age)$out
out_ind <- which(dataset$age %in% c(out))
out_ind

dataset[out_ind, ]

boxplot(dataset$age ,
        ylab= "age",
        main= "Boxplot of age")
mtext(paste("Outliers: ", paste(out, collapse = ", ")))
```
